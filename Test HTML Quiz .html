    <script>

        function checkAnswer() {

            const choices = document.getElementsByName('population');

            let selectedValue = '';

            for (const choice of choices) {

                if (choice.checked) {

                    selectedValue = choice.value;

                    break;

                }

            }


            const resultDiv = document.getElementById('result');

            if (selectedValue === '83') {

                resultDiv.innerHTML = "<strong>Correct!</strong> Germany's population is approximately 83 million as of 2024.";

                resultDiv.style.borderColor = "green";

            } else {

                resultDiv.innerHTML = "<strong>Incorrect.</strong> The correct answer is approximately 83 million.";

                resultDiv.style.borderColor = "red";

            }

            resultDiv.style.display = "block";

        }

    </script>
